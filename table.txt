CREATE TABLE Characters (
	hash char(16),
	userId char(20),
	name varchar(64),
	species varchar(64),
	stamina int,
	height int,
	mass int,
	density int,
	leftHand char(16),
	rightHand char(16),
	PRIMARY KEY (hash)
);

CREATE TABLE Item (
	hash char(16),
	userId char(20),		//discord user id
	ownerHash char(16),		//player object hash
	weight float,
	extra varchar(65536)		//Additional properties. Need a lot of string space for this.
);
